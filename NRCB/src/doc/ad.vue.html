<script>
/**
 * @module 轮播组件
 * @description 轮播图片weex组件
 * @example  <ad jsonurl="ad.json"></ad>
 * ad.json数据格式如下:
 * {
    "$schema": "ad_schema.json",
    "data": [{
            "name": "",
            "src": "imgs/tab_home/ad/banner@3x.png",
            "url": "http://www.csii.com.cn"
        },
        {
            "name": "",
            "src": "imgs/tab_home/ad/banner@3x.png",
            "url": "http://www.csii.com.cn"
        }
    ]
}
 * 
 */
const stream = weex.requireModule("stream");
const context = weex.requireModule("context");
const modal = weex.requireModule("modal");
export default {
  name:"ad",
  /**
   * Props 接受父组件的传值
   * @prop {Json} jsonurl 轮播图的json文件路径
   * 
   */
  props: ["jsonurl"],
  data() {
    return {
      ad_list: []
    };
  },
  mounted() {
    this.getAd();
  },
  methods: {
    // /**
    //  * @function goto
    //  * @param {String} url 跳转的场景名
    //  */
    goto(url) {
      context.launchStage(url);
    },
    getAd() {
      debugger;
      stream.fetch(
        {
          method: "GET",
          url: this.$store.state.jsonBaseUrl + this.jsonurl,
          type: "json"
        },
        res => {
          debugger;
          this.ad_list = res.data.data;
        }
      );
    }
  }
};
</script>
